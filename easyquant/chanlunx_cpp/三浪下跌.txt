FRAC:=TDXDLL1(2,0,H,L);{标准笔}

VAR1:=BARSLAST(FRAC=-1);
VAR2:=VAR1+REF(VAR1,VAR1+1)+1;

VAR3:=BARSLAST(FRAC=1);
VAR4:=VAR3+REF(VAR3,VAR3+1)+1;

CON1:=VAR1>=0 AND VAR2>0 AND VAR2>VAR1 AND VAR3>0 AND VAR4>0 AND VAR4>VAR3 AND VAR3>VAR1;
CON2:=REF(L,VAR1)<REF(L,VAR2);
CON3:=REF(H,VAR3)<REF(H,VAR4);

CON1 AND CON2 AND CON3;
